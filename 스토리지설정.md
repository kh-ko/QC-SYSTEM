-- 1. 'attachments' 라는 이름의 공개(Public) 파일 창고(버킷)를 생성합니다.
-- Public으로 만들면 프론트엔드에서 복잡한 인증 절차 없이 URL만으로 이미지를 바로 화면에 띄울 수 있습니다.
INSERT INTO storage.buckets (id, name, public)
VALUES ('attachments', 'attachments', true);

-- 2. 스토리지 보안 규칙(RLS) 활성화 (안전장치)
-- 파일이 저장되는 실제 테이블(storage.objects)에 대한 접근 권한을 설정합니다.
ALTER TABLE storage.objects ENABLE ROW LEVEL SECURITY;

-- 3. 조회(Select) 규칙: 누구나(Public) 이 버킷의 파일을 볼 수 있습니다. (URL로 이미지 렌더링을 위해)
CREATE POLICY "Public Access to Attachments" 
ON storage.objects FOR SELECT 
TO public 
USING ( bucket_id = 'attachments' );

-- 4. 업로드(Insert) 규칙: '로그인한 사용자(authenticated)'만 파일을 올릴 수 있습니다.
CREATE POLICY "Authenticated Users can Upload Attachments" 
ON storage.objects FOR INSERT 
TO authenticated 
WITH CHECK ( bucket_id = 'attachments' );

-- 5. 삭제(Delete) 규칙: '로그인한 사용자'만 파일을 지울 수 있습니다.
CREATE POLICY "Authenticated Users can Delete Attachments" 
ON storage.objects FOR DELETE 
TO authenticated 
USING ( bucket_id = 'attachments' );

1. 사용자가 <input type="file" /> 로 파일을 선택.

2. 프론트엔드가 Supabase Storage로 파일을 쏩니다 (supabase.storage.from('attachments').upload(...)).

3. Supabase가 파일을 저장하고 **"파일이 저장된 Public URL"**을 프론트엔드로 돌려줍니다.

4. 프론트엔드는 그 URL을 받아서, 우리가 만든 issues 테이블의 attachment_urls (TEXT 배열) 에 텍스트로 쏙 집어넣고 이슈 등록(INSERT)을 마무리합니다.
