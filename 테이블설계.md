CREATE TABLE projects (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL UNIQUE
);

-- 2. 모듈 테이블
CREATE TABLE modules (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL UNIQUE
);

-- 3. 이슈 테이블 (App 단 통제, 사용자 ID 기반)
CREATE TABLE issues (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  issue_number SERIAL UNIQUE,                  -- 자동 증가하는 이슈 번호
  project_names TEXT[] DEFAULT '{}',           -- 프로젝트 이름 배열
  module_names TEXT[] DEFAULT '{}',            -- 모듈 이름 배열
  description TEXT,                            -- 이슈 내용
  attachment_urls TEXT[] DEFAULT '{}',         -- 첨부파일 URL 배열
  assignee_ids TEXT[] DEFAULT '{}',            -- 담당자 ID 배열 (names -> ids 로 변경)
  status TEXT DEFAULT '등록',                    -- CHECK 제약조건 삭제 (App에서 통제)
  priority TEXT DEFAULT 'medium',                -- CHECK 제약조건 삭제 (App에서 통제)
  creator_id TEXT NOT NULL,                    -- 등록자 ID (name -> id 로 변경)
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 4. 대책 테이블
CREATE TABLE countermeasures (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  issue_id UUID REFERENCES issues(id) ON DELETE CASCADE, 
  apply_ver_id TEXT,                           
  description TEXT,                            
  owner_id TEXT,                               -- 대응 담당자 ID (name -> id 로 변경)
  attachment_urls TEXT[] DEFAULT '{}',         
  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
